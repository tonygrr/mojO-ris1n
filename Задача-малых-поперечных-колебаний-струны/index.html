# –î–∞–Ω–Ω—ã–µ –∑–∞–¥–∞—á–∏

–£—Ä–∞–≤–Ω–µ–Ω–∏–µ –º–∞–ª—ã—Ö —Å–≤–æ–±–æ–¥–Ω—ã—Ö –∫–æ–ª–µ–±–∞–Ω–∏–π —Å—Ç—Ä—É–Ω—ã:

$$ \dfrac{\partial^{2} U}{\partial x^{2}}=\dfrac{1}{a^{2}} \dfrac{\partial^{2} U}{\partial t^{2}}. \tag{1} $$

–ì—Ä–∞–Ω–∏—á–Ω—ã–µ —É—Å–ª–æ–≤–∏—è:

$$ U|_{x=0;l}=0, \tag{2} $$

—Ç. –µ. —Å—Ç—Ä—É–Ω–∞ –Ω–∞ –∫–æ–Ω—Ü–∞—Ö –∂—ë—Å—Ç–∫–æ –∑–∞–∫—Ä–µ–ø–ª–µ–Ω–∞ –∏ —Ñ—É–Ω–∫—Ü–∏—è –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—è –≤ —ç—Ç–∏—Ö —Ç–æ—á–∫–∞—Ö —Ä–∞–≤–Ω–∞ 0.

–ù–∞—á–∞–ª—å–Ω—ã–µ —É—Å–ª–æ–≤–∏—è (—É—Å–ª–æ–≤–∏—è –ø—Ä–∏–¥–∞–Ω–∏—è —Å–∫–æ—Ä–æ—Å—Ç–∏ —Å—Ç—Ä—É–Ω—ã –≤ –Ω–∞—á–∞–ª—å–Ω—ã–π –º–æ–º–µ–Ω—Ç –≤—Ä–µ–º–µ–Ω–∏):

$$ U|{t=0}=f(x); \quad \dfrac{\partial U}{\partial t}\bigg|{t=0}=F(x), \tag{3} $$

–≥–¥–µ $f(x)$ ‚Äî —Ñ—É–Ω–∫—Ü–∏—è, —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É—é—â–∞—è —Å—Ç–µ–ø–µ–Ω—å –Ω–∞—Ç—è–∂–µ–Ω–∏—è, $F(x)$ ‚Äî —Ñ—É–Ω–∫—Ü–∏—è, —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É—é—â–∞—è —Å—Ç–µ–ø–µ–Ω—å —É–¥–∞—Ä–∞. –û–±—ã—á–Ω–æ –æ–¥–Ω–∞ –∏–∑ –Ω–∏—Ö –±–µ—Ä—ë—Ç—Å—è —Ä–∞–≤–Ω–æ–π 0, —Ç–∞–∫ –∫–∞–∫ —É–¥–∞—Ä–∏—Ç—å –∏ –Ω–∞—Ç—è–Ω—É—Ç—å —Å—Ç—Ä—É–Ω—É –æ–¥–Ω–æ–≤—Ä–µ–º–µ–Ω–Ω–æ –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –ø—Ä–æ–±–ª–µ–º–∞—Ç–∏—á–Ω–æ.

# –ú–µ—Ç–æ–¥ –§—É—Ä—å–µ

–î–ª—è —Ä–µ—à–µ–Ω–∏—è –∑–∞–¥–∞—á–∏ –∏—Å–ø–æ–ª—å–∑—É–µ–º –º–µ—Ç–æ–¥ –§—É—Ä—å–µ (–º–µ—Ç–æ–¥ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö):

$$ U(x,t) = X(x)T(t). \tag{4} $$

–ü–æ–¥—Å—Ç–∞–≤–ª—è–µ–º (4) –≤ (1):

$$ \begin{align} X’’(x)T(t) &= \dfrac{1}{a^{2}}X(x)T’’(t); \quad | :X(x)T(t) \\\dfrac{X’’(x)}{X(x)} &= \dfrac{1}{a^{2}} \dfrac{T’’(t)}{T(t)}=-\lambda^{2}, \end{align} $$

–≥–¥–µ $\lambda^{2}$ ‚Äî –ø–æ—Å—Ç–æ—è–Ω–Ω–∞—è —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è. –†–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º –∫–∞–∂–¥–æ–µ –∏–∑ —É—Ä–∞–≤–Ω–µ–Ω–∏–π –æ—Ç–¥–µ–ª—å–Ω–æ:

$$ \begin{align} \dfrac{X’’(x)}{X(x)}+\lambda^{2}&=0; \quad | \times X(x) \\\dfrac{1}{a^{2}} \dfrac{T’’(t)}{T(t)}+\lambda^{2}&=0; \quad | \times T(t)a^{2} \\\\X’’(x)+\lambda^{2}X(x)&=0; \tag{5} \\T’’(t)+a^{2}\lambda^{2}T(t)&=0. \tag{6} \\\\\end{align} $$

–û–±—â–µ–µ —Ä–µ—à–µ–Ω–∏–µ —É—Ä–∞–≤–Ω–µ–Ω–∏–π (5) –∏ (6):

$$ \begin{align} X(x)&=c_{1}\cos(\lambda x)+c_{2}\sin(\lambda x); \tag{5.1} \\T(t)&=A\cos(\lambda at)+B\sin(\lambda at). \tag{6.1} \end{align} $$

–î–∞–ª–µ–µ —Ä–∞–±–æ—Ç–∞–µ–º —Å (5) –∏ (5.1). –ü—Ä–∏–º–µ–Ω—è–µ–º –≥—Ä–∞–Ω–∏—á–Ω—ã–µ —É—Å–ª–æ–≤–∏—è (2):

$$ X(0)=0; \quad X(l)=0. \tag{7} $$

–ü–æ–¥—Å—Ç–∞–≤–ª—è–µ–º (7) –≤ (5.1):

$$ \begin{align} c_{1}\cos(\lambda 0)+c_{2}\sin(\lambda 0) &= 0; \quad \to c_{1}=0; \\c_{1}\cos(\lambda l)+c_{2}\sin(\lambda l) &= 0; \quad \to \sin(\lambda l)=0; \\\lambda l&=\pi n; \\\lambda_{n}&=\dfrac{\pi n}{l}. \end{align} $$

–ó–∞–¥–∞—á–∞ (5) —Å –≥—Ä–∞–Ω–∏—á–Ω—ã–º–∏ —É—Å–ª–æ–≤–∏—è–º–∏ (7) ‚Äî –æ–¥–Ω–æ—Ä–æ–¥–Ω–∞—è –∫—Ä–∞–µ–≤–∞—è –∑–∞–¥–∞—á–∞. –†–µ—à–∞–µ—Ç—Å—è –æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–µ –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–µ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ (5) –ø—Ä–∏ –æ–¥–Ω–æ—Ä–æ–¥–Ω—ã—Ö –≥—Ä–∞–Ω–∏—á–Ω—ã—Ö —É—Å–ª–æ–≤–∏—è—Ö (7). –†–µ—à–µ–Ω–∏–µ (5,7):

$$ \begin{align} \begin{cases} X_{n}(x)=c_{2}\sin\left( \dfrac{\pi nx}{l} \right); \tag{8} \\T_{n}(t)=A_{n}\cos\left( \dfrac{\pi nat}{l} \right)+B_{n}\sin\left( \dfrac{\pi nat}{l} \right). \end{cases} \end{align} $$

–ü–æ–ª—É—á–∏–ª–∏ —Å–∏—Å—Ç–µ–º—É –ª–∏–Ω–µ–π–Ω—ã—Ö –∞–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–∏—Ö —É—Ä–∞–≤–Ω–µ–Ω–∏–π (8) –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω—ã—Ö –∞–º–ø–ª–∏—Ç—É–¥–Ω—ã—Ö –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–æ–≤. –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç $c_{2}$ –º–æ–∂–Ω–æ –ø—Ä–∏–Ω—è—Ç—å —Ä–∞–≤–Ω—ã–º 1, —Ç–∞–∫ –∫–∞–∫ –æ–Ω –º–æ–∂–µ—Ç –±—ã—Ç—å –ª—é–±—ã–º –∏ –Ω–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç –ø—Ä–∞–∫—Ç–∏—á–µ—Å–∫–æ–≥–æ –∏–Ω—Ç–µ—Ä–µ—Å–∞. –°–æ–≥–ª–∞—Å–Ω–æ (4):

$$ U_{n}(x,t)=X_{n}(x)T_{n}(t). \tag{9} $$

–°–æ–≥–ª–∞—Å–Ω–æ –º–µ—Ç–æ–¥—É –§—É—Ä—å–µ, —Ä–µ—à–µ–Ω–∏–µ –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –≤ –≤–∏–¥–µ –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ–≥–æ —Ä—è–¥–∞ –§—É—Ä—å–µ (–¥–∏—Å–∫—Ä–µ—Ç–Ω–æ–≥–æ —Å–ø–µ–∫—Ç—Ä–∞):

$$ \boxed{ \begin{align} U(x,t)&=\sum_{n=1}^{\infty}U_{n}(x,t) \\&=\sum_{n=1}^{\infty}\left( A_{n}\cos\left(\dfrac{\pi ant}{l}\right)+B_{n}\sin \underbrace{\left( \dfrac{\pi ant}{l}\right)}{\omega{n}} \right)\sin\left( \dfrac{\pi nx}{l} \right). \tag{10} \end{align} } $$

–î–∞–ª–µ–µ —Å—Ç–∞–≤–∏—Ç—Å—è –∑–∞–¥–∞—á–∞ –®—Ç—É—Ä–º–∞-–õ–∏—É–≤–∏–ª–ª—è, —Å–æ–≥–ª–∞—Å–Ω–æ –∫–æ—Ç–æ—Ä–æ–π —É—Ä–∞–≤–Ω–µ–Ω–∏–µ (8) ‚Äî —Å–æ–±—Å—Ç–≤–µ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è, –∞

$$ \lambda_{n}^{2}=\left( \dfrac{\pi n}{l} \right)^{2} $$

‚Äî —Å–æ–±—Å—Ç–≤–µ–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ.

–£—Ä–∞–≤–Ω–µ–Ω–∏–µ (10) —É–¥–æ–≤–ª–µ—Ç–≤–æ—Ä—è–µ—Ç –≥—Ä–∞–Ω–∏—á–Ω—ã–º —É—Å–ª–æ–≤–∏—è–º (2), –Ω–æ –Ω–µ —É–¥–æ–≤–ª–µ—Ç–≤–æ—Ä—è–µ—Ç –Ω–∞—á–∞–ª—å–Ω–æ–º—É —É—Å–ª–æ–≤–∏—é (3). –ü–æ–¥—Å—Ç–∞–≤–∏–º (3) –≤ (10):

$$ \begin{align} \begin{cases} f(x)=\sum_{n=1}^{\infty}A_{n}\sin\left( \dfrac{\pi nx}{l} \right); \\F(x)=\sum_{n=1}^{\infty}B_{n} \dfrac{\pi an}{l}\sin\left( \dfrac{\pi nx}{l} \right); \end{cases} \quad \bigg | \times \sin\left( \dfrac{\pi nx}{l} \right);\int_{0}^{1} , dx \tag{11} \end{align} $$

–°–º—ã—Å–ª –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–π –Ω–∞–¥ —Å–∏—Å—Ç–µ–º–æ–π (11): –∏—Å–ø–æ–ª—å–∑—É—è —Å–≤–æ–π—Å—Ç–≤–æ –æ—Ä—Ç–æ–≥–æ–Ω–∞–ª—å–Ω–æ—Å—Ç–∏, –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç—Å—è –¥–æ–º–Ω–æ–∂–µ–Ω–∏–µ –∫–∞–∂–¥–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è –Ω–∞ —Å–æ–±—Å—Ç–≤–µ–Ω–Ω—É—é —Ñ—É–Ω–∫—Ü–∏—é –∏ –∏–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏–µ –≤ –ø—Ä–µ–¥–µ–ª–∞—Ö –æ—Ç 0 –¥–æ 1. –í —Ä–µ–∑—É–ª—å—Ç–∞—Ç–µ –ø–æ–ª—É—á–∞–µ—Ç—Å—è –±–µ—Å–∫–æ–Ω–µ—á–Ω–∞—è —Å—É–º–º–∞ –∏–∑ –∏–Ω—Ç–µ–≥—Ä–∞–ª–æ–≤, –Ω–æ –æ—Ç –≤—Å–µ–π —ç—Ç–æ–π —Å—É–º–º—ã –æ—Å—Ç–∞—ë—Ç—Å—è –ª–∏—à—å –æ–¥–Ω–æ —Å–ª–∞–≥–∞–µ–º–æ–µ: –ø—Ä–∏ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–∏ $n$ –≤ —Å—É–º–º–µ —Å $n$, –Ω–∞ –∫–æ—Ç–æ—Ä–æ–µ –º—ã –¥–æ–º–Ω–æ–∂–∞–µ–º. –≠—Ç–æ —Å–ª–∞–≥–∞–µ–º–æ–µ ‚Äî –Ω–µ–Ω—É–ª–µ–≤–æ–π –∏–Ω—Ç–µ–≥—Ä–∞–ª, –∫–æ—Ç–æ—Ä—ã–π –º–æ–∂–Ω–æ –Ω–∞–±–ª—é–¥–∞—Ç—å –≤ —Å–∏—Å—Ç–µ–º–µ (12).

$$ \boxed{ \begin{align} \begin{cases} \displaystyle A_{n}=\dfrac{2}{l}\int_{0}^{l} f(x)\sin\left( \dfrac{\pi nx}{l} \right) , dx; \\\displaystyle B_{n}=\dfrac{2}{\pi na}\int_{0}^{l} F(x)\sin\left( \dfrac{\pi nx}{l} \right) , dx. \end{cases} \tag{12} \end{align} } $$
